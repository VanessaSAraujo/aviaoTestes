package org.example.services;

import java.util.regex.Pattern;

public class PassengerService {
    public int soma;
    public int primeiroDigitoVerificador;
    public int segundoDigitoVerificador;

    public boolean validarCpf(String cpf) {
        cpf = cpf.replaceAll("\\D", "");
        if (cpf.length() != 11 || cpf.matches("(\\d)\\1{10}")) return false;

        soma = 0;
        for (int i = 0; i < 9; i++) {
            soma += (cpf.charAt(i) - '0') * (10 - i);
        }
        primeiroDigitoVerificador = 11 - (soma % 11);
        primeiroDigitoVerificador = primeiroDigitoVerificador >= 10 ? 0 : primeiroDigitoVerificador;
        if (primeiroDigitoVerificador != cpf.charAt(9) - '0') return false;

        soma = 0;
        for (int i = 0; i < 10; i++) {
            soma += (cpf.charAt(i) - '0') * (11 - i);
        }
        segundoDigitoVerificador = 11 - (soma % 11);
        segundoDigitoVerificador = segundoDigitoVerificador >= 10 ? 0 : segundoDigitoVerificador;
        return segundoDigitoVerificador == cpf.charAt(10) - '0';
    }

    public boolean validarEmail(String email) {
        String padraoEmail = "^[\\w\\.-]+@[\\w\\.-]+\\.\\w+$";
        return Pattern.matches(padraoEmail, email);
    }
}
